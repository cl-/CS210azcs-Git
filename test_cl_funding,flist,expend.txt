public class Funding {
    //2 Fields, 2 inputEnabled
    public int amount;
    public String source;
    
    //2 Constructors
    public Funding () {}
    
    public Funding (int inp_amount, String inp_source) {
        amount = inp_amount; source = inp_source;
    }
    
    //Methods
    public String toString() {return amount+ ' ' +source+ "\n";}
    public String forPrint() {return "$" +amount+ " by " +source+ "\n";}

    public int    getAmount(){return amount;}
    public String getSource(){return source;}
}


public class FundingList extends ArrayList<Funding> {
    //1 Fields, 1 inputEnabled
    public int amountTotal;
    public String sourceEvery;
    public int needRecalc;

    //1 Constructor since it is an arrayList.
    public FundingList () {}

    //Methods
    public String toString() {return amountTotal+ ' ' +sourceEvery+ "\n";}
    public String forPrint() {return toDollarSign(amountTotal)+ " by " +sourceEvery+ "\n";}
    
    public void addFunding(){recalcAll();}

    public void recalcAll() {recalcAmounts(); recalcSources();}

    public int recalcAmounts(){
       amountTotal=0;
       for (int i=0; i<FundingList.size(); ++i)
       {amountTotal += FundingList[i].getAmount();}
       return amountTotal;  }
    public String recalcSources(){
       sourceEvery='';
       for (int i=0; i<FundingList.size(); ++i)
       {sourceEvery += FundingList[i].getSource()+' ';}
       return sourceEvery;  }
    
    public int getAmountTotal(){return amountTotal;}
    public String getSourceEvery(){return sourceEvery;}
}
